244  public static string ExceptionToDetailedString(string message, params Exception[] ex) {
245  StringBuilder text = new StringBuilder(message == null || message.Length == 0 ? "=>An error occurred on " + DateToString(DateTime.Now) + ".\n" : "=>An error occurred on " + DateToString(DateTime.Now) + ". \n\nException details:\n\n " + message + "\n");

276  toSerialize.Remove("InnerExceptions");
277  text.Append("\n\nJson: " + Regex.Unescape(JsonConvert.SerializeObject(ToDynamic(toSerialize), Formatting.Indented)) + "\n");
278  } catch {
279  try {
280  temp = currentException.Message;
281  if (temp != null) {
282  temp = temp.Trim();
283  if (temp.Length != 0) {
284  text.Append(":\n\nMessage:\n\n ");

287  }

290  if (temp.Length != 0) {
