110  private static object[] PrepareParams(object[] items) {
111  if (items == null || items.Length == 0)
112  return null;
113  object[] fixedItems = new object[items.Length];
114  object current;
115  for (int i = 0; i < items.Length; i++) {
116  current = items[i];
117  if (current == null)
118  fixedItems[i] = null;
119  else if (current is Contract)
120  fixedItems[i] = ((Contract) current).Address;
121  else if (current is DeployedContract)
122  fixedItems[i] = ((DeployedContract) current).Contract.Address;
123  else if (current is Wallet)
124  fixedItems[i] = ((Wallet) current).Address;
125  else if (current is TransactionReceipt)
126  fixedItems[i] = ((TransactionReceipt) current).TransactionHash;
127  else
128  fixedItems[i] = items[i];
129  }
130  return fixedItems;
131  }
